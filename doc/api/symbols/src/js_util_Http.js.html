 23 /**
 24  * HTTP Ajax Helper.
 25  *
 26  * @requires jQuery
 27  * @class
 28 */
 29 khtml.maplib.util.Http = function() {
 30 	/**
 31 	 * Load data from the server using a HTTP GET request.
 32 	 *
 33 	 * @see http://api.jquery.com/jQuery.get/
 34 	 * @param {String}	url	A string containing the URL to which the request is sent.
 35 	 * @param {Object/String} data	An object or string that is sent to the server with the request.
 36 	 * @param {function} successHandler successHandler(data, textStatus, jqXHR) - A callback function that is executed if the request succeeds.
 37 	 * @param {String} dataType	The type of data expected from the server. Default: Intelligent Guess (xml, json, script, or html).
 38 	*/
 39 	this._get = function(url, data, successHandler, dataType) {
 40 		try {
 41 			return jQuery.get(url,data,successHandler,dataType);
 42 		} catch(e) {
 43 			khtml.maplib.base.Log.warn("Exception occured! Did you include the jQuery library?");
 44 			khtml.maplib.base.Log.exception(e);
 45 		}
 46 	};
 47 	
 48 	/**
 49 	 * Decorates {khtml.maplib.util.Http._get} with a method that writes a log entry if the request failes.
 50 	 *
 51 	 * @see _get
 52 	 * @see http://api.jquery.com/jQuery.get/
 53 	 * @param {String}	url	A string containing the URL to which the request is sent.
 54 	 * @param {Object/String} data	An object or string that is sent to the server with the request.
 55 	 * @param {function} successHandler successHandler(data, textStatus, jqXHR) - A callback function that is executed if the request succeeds.
 56 	 * @param {String} dataType	The type of data expected from the server. Default: Intelligent Guess (xml, json, script, or html).
 57 	*/
 58 	this.get = function(url, data, successHandler, dataType) {
 59 		var jqxhr = this._get(url, data, successHandler, dataType)
 60 	  		.error(function(data) { khtml.maplib.base.Log.error("khtml.maplib.util.Http.get(): request failed!", data) });
 61 	
 62 		return jqxhr;
 63 	};	
 64 };
