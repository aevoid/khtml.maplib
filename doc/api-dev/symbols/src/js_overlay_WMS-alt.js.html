 23 /**
 24  * WMS Overlay.
 25  *
 26  * Overlay the map with a layer from a wms server.
 27  * Only WGS84 (EPSG:4326) is supported.
 28  * WMS URL's are required to include '&SRS=EPGSG:4326'.
 29  *
 30  * WMS Specification: http://www.opengeospatial.org/standards/wms
 31  *
 32  * Example:
 33 	<pre>
 34 	var wms = new khtml.maplib.overlay.WMS(
 35 		{
 36 			opacity: 0.6, 
 37 			url: 'http://wms.geo.admin.ch/',
 38 			parameters: {'FORMAT':'image/png', 'STYLES': 'default'},
 39 			layers: ['xyz','abc']
 40 		}
 41 	);
 42 	map.addOverlay(wms);
 43 	</pre>
 44  *
 45  * @param {Object} options Properties: 'url' => url to wms server, use as <img src="url">. The following parameters are appended to the url: '&BBOX=...&WIDTH=...&HEIGHT=...&xyz=...'. 'opacity' => transparency of image where 0 is fully transparent and 1 is not transparent. 'class' => class is used for the image tag.
 46  * @class
 47  * @borrows khtml.maplib.overlay.RenderableMixin#_parseOptions as #_parseOptions
 48  * @borrows khtml.maplib.overlay.RenderableMixin#parent as #parent
 49  * @borrows khtml.maplib.overlay.RenderableMixin#map as #map
 50 */
 51 khtml.maplib.overlay.WMS = function(options) {	
 52 	// ---------------------------------------------------------------
 53 	// Private - should not be used from the outside
 54 	// ---------------------------------------------------------------
 55 		
 56 	this._createRenderElement = function() {
 57 		// create container div
 58 		var innerDiv = document.createElement("div");
 59 		
 60 		// create image 
 61 		var img = document.createElement("img");
 62 		img.style.opacity=this.opacity();//0.5;
 63 		if(this.className() != null) { img.className = this.className();}
 64 		
 65 		// store reference to image
 66 		this.img = img;
 67 		
 68 		// image load event
 69 		khtml.maplib.base.helpers.eventAttach(this.img, "load", this._imgLoaded, this, false);
 70 
 71 		// apped image to container
 72 		innerDiv.appendChild(img);
 73 		
 74 		// style innerDiv - set defaults
 75 		innerDiv.style.position = "absolute";
 76 		innerDiv.style.top = "0px";
 77 		innerDiv.style.left = "0px";
 78 		
 79 		return innerDiv;
 80 	}
 81 
 82 	/**
 83 	 * Reset position of image to top-left corner after new image is loaded.
 84 	 * Resets the image position after the image has been moved.
 85 	*/
 86 	this._imgLoaded = function() {
 87 		this.renderElement.style.top = "0px";
 88 		this.renderElement.style.left = "0px";
 89 		this.img.width = this.img.clientWidth;
 90 		this.img.height = this.img.clientHeight;
 91 		this.bounds = null;
 92 	}
 93 	
 94 	/**
 95 	 * Calculate URL for WMS image request.
 96 	*/
 97 	this._calculateUrl = function() {
 98 		// --- WMS Calculation ---
 99 		// 1. get center position of map
100 		// 2. get width and height of map
101 		// 3. get zoomlevel
102 		// 4. calculate meter per pixel
103 		//		- based on zoomlevel
104 		// 5. calculate meter per width and height
105 		//		- width * meter per pixel
106 		//		- height * meter per pixel
107 		// 6. convert width/height in meter to distance in wgs84 lat/lng
108 		// 7. calculate wms BBOX
109 		//		- bottomleft: center - wgs84distance/2
110 		//		- topright: center + wgs84distance/2
111 		// --- End WMS Calculation ---
112 
113 		// 1.
114 		var mapCenter = this.map().center();
115 		var x = mapCenter.lat();
116 		var y = mapCenter.lng();
117 		// 3.
118 		var z = this.map().getIntZoom();
119 		
120 		// 2.
121 		var imgwidth = this._map.mapsize.width;
122 		var imgheight = this._map.mapsize.height;
123 		
124 		// OR! JUST SKIP THE CALCULATION AND GET LAT/LNG FROM THIS.MAP() ;-)
125 		
126 		// 7.
127 		var bottomLeftPoint = this.map().XYTolatlng(0,imgheight);
128 		var topRightPoint = this.map().XYTolatlng(imgwidth,0);		
129 
130 		lng = bottomLeftPoint.lng();
131 		lat = bottomLeftPoint.lat();
132 		nextLng = topRightPoint.lng();
133 		nextLat = topRightPoint.lat();
134 		
135 		this.bounds = new khtml.maplib.geometry.Bounds(bottomLeftPoint,topRightPoint);
136 
137 		// TODO: if I add xyz parameter in the imagecall all works well.
138 		//			Without it dosen't. It works with some failures but not allways
139 		
140 		var params = "&BBOX="+lng+","+lat+","+nextLng+","+nextLat+"&WIDTH="+imgwidth+"&HEIGHT="+imgheight+"&xyz="+x+","+y+","+z;
141 		var url = this.wmsUrl() + params;
142 		
143 		return url;
144 	}
145 	
146 	/**
147 	 * Set new image url - but only modify "src"-attribute if the url has changed,
148 	 * prevent's reload of the same image.
149 	*/
150 	this._updateImage = function() {	
151 		if(this._newUrl!=this.img.src) {
152 			this.img.src=this._newUrl;
153 			khtml.maplib.base.Log.debug('WMS image url:',this._newUrl);
154 		}
155 	}
156 	
157 	// ---------------------------------------------------------------
158 	// Public
159 	// ---------------------------------------------------------------
160 
161 	/**
162 	 * Initialize
163 	 *
164 	 * @params {Object} options Use options.map and options.parent to assign map/parent. Example: <pre>init({map:myMap, parent: myParent});</pre>
165 	*/
166 	this.init = function(options) {
167 		this._parseOptions(options);
168 	}
169 		
170 	/**
171 	 * Render this WMS Overlay.
172 	 *
173 	 * 1. ensure that this Layer has been initialized. (this.init()).
174 	 * 2. execute all pending dom manipulations (this._updateDom()).
175 	 * 3. execute render() on contained layers and features.
176 	*/
177 	this.render = function() {
178 		if (!this.renderElement || !this._map || !this._url) {
179 			khtml.maplib.base.Log.warn('Error: WMS.render() - not all properties are initialized (requires renderElement,_map,_url)!');
180 			return false;
181 		}
182 
183 		// Attach render element to parent
184 		this._updateDom();
185 				
186 		// Don't render while moving image.
187 		if(this.moving) {
188 			// Drag&drop is way faster than the WMS server's ability to generate images.
189 			// Wait for the final position (when dragging has stopped) to request a new image.
190 			// TODO: Same is true for zooming - using the workaround below for zooming.
191 			return false;
192 		}
193 		
194 		var url = this._calculateUrl();
195 		
196 		// WORKAROUND Zooming
197 		// Problem: to give a smooth zooming impression the map is rendered in up to 20 steps
198 		//			between start-zoomlevel and end-zoomlevel.
199 		// 			each of this temporary zoomlevel calls WMS.render().
200 		//			a temporary zoomlevel is displayed a small amount of time - 
201 		//			you can't fetch a new WMS image in this time and display it - it's too fast!
202 		//			Too reduce load on the WMS server, those temporary-zoomlevel images are not requested.
203 		//			To achieve this, a 100ms timeout is introduced - as long as temporary render() calls are made
204 		//			the timeout is cleared before it can be executed (clearTimeout).
205 		//			As a result - only 1 request - the last request - is executed (like we wanted :).
206 		if(this._newUrl == '') {
207 			// first call - skip timeout
208 			this._newUrl = url;
209 			this._updateImage();
210 		} else {
211 			this._newUrl = url;
212 
213 			clearTimeout(this._updateImageFunc);
214 			that = this;
215 			var func = function() {
216 				that._updateImage();
217 			};
218 			var timeout = 100;
219 			this._updateImageFunc = setTimeout(func, 100);
220 		}
221 		// END WORKAROUND Zooming
222 	}
223 
224 	/**
225 	 * Remove element from DOM.
226 	*/
227 	this.clear = function() {
228 		this._removeChildRenderElementFromParent(this);
229 	}
230 	
231 	// Moveable extension
232 	// Required for map dragging
233 	this.makeMoveable = function() {
234 		khtml.maplib.base.helpers.eventAttach(this.renderElement, "mousedown", this.down, this, false);
235 		khtml.maplib.base.helpers.eventAttach(window, "mousemove", this.move, this, true);
236 		khtml.maplib.base.helpers.eventAttach(window, "mouseup", this.up, this, true);
237 	}
238 	this.moving = false;
239 	this.dx = 0;
240 	this.dy = 0;
241 	this.down = function(evt) {
242 		this.moving = true;
243 		this.dx = this.renderElement.offsetLeft - this._map.pageX(evt)
244 		this.dy = this.renderElement.offsetTop - this._map.pageY(evt)
245 	}
246 	this.move = function(evt) {
247 		if (this.moving) {
248 			var x = this._map.pageX(evt) + this.dx;
249 			var y = this._map.pageY(evt) + this.dy;
250 			this.renderElement.style.left=x+"px";
251 			this.renderElement.style.top=y+"px";
252 		}
253 	}
254 	this.up = function() {
255 		this.moving = false;
256 		this.render();
257 	}
258 
259 	// ------------------------------------------------------------
260 	// Setter's & getter's
261 	// ------------------------------------------------------------
262 
263 	this.opacity = function(arg1) {
264 		if (typeof(arg1)!='undefined') {
265 			this._opacity = parseFloat(arg1);
266 		}
267 		
268 		return this._opacity;
269 	}
270 	
271 	this.url = function(arg1) {
272 		if (typeof(arg1)!='undefined') {
273 			this._url = arg1;
274 		}
275 		
276 		return this._url;
277 	}
278 	
279 	this.className = function(arg1) {
280 		if (typeof(arg1)!='undefined') {
281 			this._class = arg1;
282 			if(typeof(this.img)!='undefined') {
283 				this.img.className = this._class;
284 			}
285 		}
286 		
287 		return this._class;
288 	}
289 	
290 	// ?SERVICE=WMS&REQUEST=GetMap&VERSION=1.1.0&LAYERS=ch.blw.ursprungsbezeichnungen-fleisch
291 	// &STYLES=default&FORMAT=image/png&SRS=EPSG:4326
292 	this.parameters = function(arg1) {
293 		if (typeof(arg1)!='undefined') {
294 			// copy parameters
295 			if(typeof(arg1)=='object'){
296 				for(var x in arg1) {
297 					this._parameters[(x).toUpperCase()] = arg1[x];
298 				}
299 			}
300 		}
301 		
302 		return this._parameters;
303 	}
304 	
305 	this.layers = function(arg1) {
306 		if (typeof(arg1)!='undefined') {
307 			if(arg1 instanceof Array) {
308 				this._layers = arg1;
309 			}
310 		}
311 		
312 		return this._layers;
313 	}
314 	
315 	this.wmsUrl = function(){
316 		var url = this.url();
317 		var sep = '?';
318 		for(var x in this._parameters) {
319 			url += sep+(x).toUpperCase()+'='+escape(this._parameters[x]);
320 			
321 			if('?'==sep)sep='&';
322 		}
323 		
324 		url += sep + 'LAYERS=' + escape(this._layers.join(','));
325 		
326 		return url;
327 	}
328 	
329 	// ------------------------------------------------------------
330 	// Constructor
331 	// ------------------------------------------------------------
332 		
333 	// Mixins
334 	khtml.maplib.base.helpers.mixin(
335 		khtml.maplib.overlay.WMS.prototype, 
336 		khtml.maplib.overlay.RenderableMixin
337 	);
338 
339 	// Elements
340 	this.parent(null);
341 	this.map(null);
342 	this.opacity(1);
343 	this.url(null);
344 	this.className(null);
345 	this._newUrl = '';
346 	this._updateImageFunc = null;
347 	
348 	// Default parameters
349 	this._parameters = {
350 		'SERVICE': 'WMS',
351 		'REQUEST': 'GetMap',
352 		'VERSION': '1.1.0',
353 		//'STYLES': '',
354 		'FORMAT': 'image/png',
355 		'SRS': 'EPSG:4326'		// only supported format
356 	};
357 	this._layers = new Array();
358 	// ?SERVICE=WMS&REQUEST=GetMap&VERSION=1.1.0&LAYERS=ch.blw.ursprungsbezeichnungen-fleisch
359 	// &STYLES=default&FORMAT=image/png&SRS=EPSG:4326
360 	
361 		
362 	// Flags
363 	this._requiresUpdateParent = true;
364 
365 	// Arguments
366 	this.options = options;
367 	if(options.className) { this.className(options.className); }
368 	if(options.url) { this.url(options.url); }
369 	if(options.parameters) { this.parameters(options.parameters); }
370 	if(options.layers) { this.layers(options.layers); }
371 	if(typeof(options.opacity)!='undefined') { this.opacity(options.opacity); }
372 
373 	this.renderElement = this._createRenderElement();
374 	this.makeMoveable(); // enable dragging of the wms overlay
375 }
