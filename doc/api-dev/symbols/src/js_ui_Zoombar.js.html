 23 //this class expect 2 elements with specific classes in the map div
 24 //zoombar and scrollhandle
 25 /*
 26  <div id="map">
 27  <div class="zoombar" style="position:absolute;width:30px;height:221px;overflow:hidden">
 28  <div class="scrollhandle" style="position:absolute;left:0px;top:0px;width:30px;height:300px;background-color:yellow"> </div>
 29  <img class="scrollbar " src="scrollbar.png" style="position:absolute;left:0px;top:00px;" />
 30  </div>
 31  */
 32 
 33 /**
 34  * Zoombar
 35  *
 36  * @class
 37 */
 38 khtml.maplib.ui.Zoombar = function() {
 39 	// -------------------------------
 40 	// Private Methods.
 41 	// Intended to be accessed internaly only
 42 	// -------------------------------
 43 	
 44 	/**
 45 	 * Cancels the event if it is cancelable, 
 46 	 * without stopping further propagation of the event.
 47 	*/
 48 	this._cancelEvent = function(evt) {
 49 		evt.cancelBubble = true;
 50 		if (evt.stopPropagation)
 51 			evt.stopPropagation();
 52 	}
 53 	/**
 54 	 * Prevents further propagation of the current event.
 55 	*/
 56 	this._stopEventPropagation = function(evt) {
 57 		if (evt.preventDefault) {
 58 			evt.preventDefault(); // The W3C DOM way
 59 		} else {
 60 			evt.returnValue = false; // The IE way
 61 		}		
 62 	}
 63 	/**
 64 	 * Calculate Z if Y is given. (also use in testcase)
 65 	*/
 66 	this._calcZFromY = function(y) {
 67 		return 22 - y / 10;
 68 	}
 69 	/**
 70 	 * Zoom
 71 	*/
 72 	this._zoom = function(evt) {
 73 		var y = this.themap.pageY(evt);
 74 		this.themap.zoom(this._calcZFromY(y));
 75 	}
 76 	
 77 	// -------------------------------
 78 	// Public Methods.
 79 	// Intended to be called from the outside.
 80 	// -------------------------------
 81 	
 82 	/**
 83 	 * called by maplib once
 84 	 */
 85 	this.init = function(themap) {
 86 		this.themap = themap;
 87 		var els = themap.mapParent.getElementsByTagName("*");
 88 		for ( var i = 0; i < els.length; i++) {
 89 			var el = els.item(i);
 90 			if (el.className == "scrollhandle") {
 91 				this.scrollhandle = el;
 92 			}
 93 			if (el.className == "zoombar") {
 94 				this.zoombar = el;
 95 			}
 96 		}
 97 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "mousedown", this.down, this, false);
 98 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "mousemove", this.move, this, false);
 99 		khtml.maplib.base.helpers.eventAttach(this.zoombar, "mouseup", this.up, this, false);
100 	}
101 	/**
102 	 * called by maplib on every map change
103 	 */
104 	this.render = function() {
105 		var top = (22 - this.themap.zoom()) * 10;
106 		this.scrollhandle.style.marginTop = top + "px";
107 	}
108 
109 	/**
110 	 * Mouse DOWN
111 	*/
112 	this.down = function(evt) {
113 		this._cancelEvent(evt);
114 		
115 		this.moving = true;
116 		
117 		this._zoom(evt);
118 		
119 		this._stopEventPropagation(evt);
120 	}
121 	/**
122 	 * Called while scrolling.
123 	*/
124 	this.move = function(evt) {
125 		this._cancelEvent(evt);
126 		
127 		if (this.moving) {
128 			this._zoom(evt);
129 			
130 			this._stopEventPropagation(evt);
131 		}
132 	}
133 	/**
134 	 * Mouse UP
135 	*/
136 	this.up = function(evt) {
137 		this._cancelEvent(evt);
138 		
139 		this.moving = false;
140 		
141 		this._stopEventPropagation(evt);
142 	}
143 }
