 34 khtml.maplib.overlay.Vector = function() {
 35 	this.vectorPointsPerIteration=2000;
 36 	this.lineArray = new Array();
 37 	this.stopit = false;
 38 	this.backend = "svg";
 39 	if (navigator.userAgent.indexOf("Android") != -1) {
 40 		this.backend = "canvas";
 41 	}
 42 	this.svgStyleInterface = false;
 43 	if (navigator.userAgent.indexOf("MSIE") != -1) {
 44 		if (getInternetExplorerVersion() < 9) {
 45 			this.backend = "vml";
 46 		}
 47 	}
 48 	this.canvasClassStyles = new Array();
 49 
 50 	this.boundsSouth = 90;
 51 	this.boundsNorth = -90;
 52 	this.boundsWest = 180;

 55 	/**
 56 	 * Switch render backend.
 57 	 * Previous rendered content (the old overlay) is discarded!
 58 	 *
 59 	 * @param {String}	render	svg, canvas or vml
 60 	*/
 61 	this.renderbackend = function(render) {
 62 		if(render){
 63 			this.backend = render;
 64 			if(this.vectorEl && this.vectorEl.parentNode){
 65 				var parent=this.vectorEl.parentNode;
 66 				parent.removeChild(this.vectorEl);
 67 			}
 68 			var vectorEl = this.createVectorElement(this.themap);
 69 			parent.appendChild(this.vectorEl);
 70 			this.render();
 71 		}
 72 		return this.backend;
 73 	}

 75 	/**
 76 	 * Create a vector element, depending on the active backend (svg, canvas or vml)
 77 	 *
 78 	 * @param {khtml.maplib.base.Map} themap reference to mapobject
 79 	*/
 80 	this.createVectorElement = function(themap) {
 81 		switch (this.backend) {
 82 		case "canvas":
 83 			vectorEl = document.createElement("canvas");
 84 			this.ctx = vectorEl.getContext("2d");
 85 			break;
 86 		case "svg":
 87 			vectorEl = document.createElementNS("http://www.w3.org/2000/svg",
 88 					"svg");
 89 			break;
 90 		case "vml":
 91 			if(document.namespaces){
 92 			if (document.namespaces['v'] == null) {
 93 				document.namespaces.add("v", "urn:schemas-microsoft-com:vml");
 94 				var stl = document.createStyleSheet();
 95 				stl.addRule("v\\:group", "behavior: url(#default#VML);");
 96 				stl.addRule("v\\:polyline", "behavior: url(#default#VML);");
 97 				stl.addRule("v\\:stroke", "behavior: url(#default#VML);");
 98 				stl.addRule("v\\:fill", "behavior: url(#default#VML);");
 99 			}
100 			}
101 			//vectorEl=document.createElement("v:group");
102 			vectorEl = document.createElement("div");
103 			//document.body.appendChild(vectorEl);
104 			//vectorEl.style.display="none";
105 			break;
106 		default:
107 			alert("error: unknown vector backend");
108 
109 		}
110 		vectorEl.style.width = themap.mapsize.width + "px";
111 		vectorEl.style.height = themap.mapsize.height + "px";
112 		vectorEl.setAttribute("height", themap.mapsize.height + "px");
113 		vectorEl.setAttribute("width", themap.mapsize.width + "px");
114 		vectorEl.style.position = "absolute";
115 		vectorEl.style.top = "0";
116 		vectorEl.style.left = "0";
117 		return vectorEl;
118 	}

157 	this.createPolyline = function(polyline) {
158 		polyline.tags = new Array;
159 		if(!polyline.close){
160 			polyline.close = false;
161 		}
162 		polyline.id = null;
163 		if(!polyline.geometry.coordinates){
164 				polyline.geometry.coordinates=new Array();
165 		}	
166 		
167 		/*

196 	this.render = function(a) {
197 		var intZoom = Math.floor(this.themap.zoom());
198 		if(!this.vectorEl)return;
199 		this.vectorEl.setAttribute("class", "z" + intZoom);
200 		if (this.stopit) {
201 			//stop rendering vectors
202 			this.stopit = false;
203 			if (a != null) {
204 				if (this.oldVectorEl && this.oldVectorEl.parentNode) {
205 					//draw not finished - go back to old version (move)
206 					try { //ie workaround
207 						this.oldVectorEl.parentNode.removeChild(this.oldVectorEl);

238 						console.log("move not passible");		
239 					}
240 				}
241 
242 				if (!this.themap.finalDraw) {
243 					return;
244 				}
245 				//the oldVectorEl will be visible until the new vectorEl is completely redered. 
246 				this.oldVectorEl = this.vectorEl;
247 				this.vectorEl = this.createVectorElement(this.themap);
248 			}else{
249 				this.clear();
250 				//these values are nessessary for vector rendering optimization
251 				this.oldtimestamp=new Date().getTime();
252 				this.starttimestamp=new Date().getTime();
253 				this.totalPointsCounter=0;
254 				this.totalLinesCounter=0;
255 				this.renderCount=0;
256 			}
257 			this.oldZoom = this.themap.zoom();
258 			//this.clear();
259 			var a = this.lineArray.length -1;
260 
261 			this.lastMoveX = this.themap.moveX;
262 			this.lastMoveY = this.themap.moveY;
263 		}
264 
265 		if (a < 0) {
266 			if(this.backend=="canvas"){
267 				this.ctx.stroke();
268 				this.ctx.fill();
269 			}
270 			this.timestamp=new Date().getTime();
271 			this.renderIterationTime=this.timestamp - this.starttimestamp;
272 			this.totalPoints=this.totalPointsCounter;
273 			this.totalLines=this.totalLinesCounter;
274 			//rendering is finished
275 			if (this.oldVectorEl && this.oldVectorEl.parentNode) {
276 				//use action was move without zoom	
277 				this.oldVectorEl.parentNode.replaceChild(this.vectorEl, this.oldVectorEl);
278 				this.vectorEl.style.display = "";
279 				return;
280 			}
281 			return;
282 		}
283 		if(typeof(a)=="number"){
284 			var line=this.lineArray[a];
285 		}else{

311 
312 					}else{

314 					}

316 					line.path=path;

319 					var path=line.path;
320 				}
321 				break;
322 			case "vml":
323 				var path = document.createElement("v:polyline");
324 				var fillEl = document.createElement("v:fill");
325 				path.appendChild(fillEl);
326 				var strokeEl = document.createElement("v:stroke");
327 				path.appendChild(strokeEl);
328 			
329 				path.setAttribute("fillcolor", fill);
330 				path.setAttribute("strokecolor", stroke);

336 			//calculate polyline path

338 				case "svg":

352 				case "vml":
353 					var ret=khtml.maplib.overlay.renderer.VML._calculatePath(line.geometry.coordinates,line.geometry.type,this.themap);

374 					path.setAttribute("points",ret.d);
375 					if(line.geometry.type=="Polygon" || line.geometry.type=="MultiPolygon"){
376 						path.setAttribute("filled",true);
377 					}else{
378 						path.setAttribute("filled",false);
379 					}
380 
381 					this.vectorEl.appendChild(path);
382 					
383 					break;
384 				default:
385 					console.log("error: unknow backend");
386 			}
387 
388 			this.renderCount+=ret.points;
389 			this.totalPointsCounter+=ret.points;
390 			this.totalLinesCounter+=ret.lines;
391 
392 		} //end inside map bounds
393 
394 		//cancel able after some lines
395 //		this.canvasRenderTimeout=null;
396 		if(typeof(a)!="number"){
397 			if(this.backend=="canvas"){

409 		}
410 		//if (a / 1000 == Math.floor(a / 1000)) {
411 		if (this.renderCount > this.vectorPointsPerIteration) {
412 			this.timestamp=new Date().getTime();
413 			this.iterationTime=this.timestamp - this.oldtimestamp;

416

418 			this.renderCount=0;
419 			var that = this;

421 				that.render(a - 1);
422 			}
423 			setTimeout(tempFunction, 0);
424 			if(this.backend=="canvas"){
425 				this.ctx.stroke();

427 				this.ctx.beginPath();
428 			}

430 			this.render(a - 1);

438 	this.makeBounds=function(coordinates){
439 		var boundsSouth = 90;
440 		var boundsNorth = -90;
441 		var boundsWest = 180;
442 		var boundsEast = -180;
443 
444 		for(var i=0; i < coordinates.length;i++){
445 			if(coordinates[i] instanceof khtml.maplib.LatLng){
446 				var lng = coordinates[i].lng();
447                                 var lat = coordinates[i].lat();
448                                 if (lat > boundsNorth) {
449                                         boundsNorth = lat;
450                                 }
451                                 if (lat < boundsSouth) {
452                                         boundsSouth = lat;
453                                 }
454                                 if (lng > boundsEast) {
455                                         boundsEast = lng;
456                                 }
457                                 if (lng < boundsWest) {
458                                         boundsWest = lng;
459                                 }
460 
461 			}else{
462 				var bounds=this.makeBounds(coordinates[i]);
463 				if(bounds.sw().lat() < boundsSouth){
464 					boundsSouth=bounds.sw().lat();
465 				}
466 				if(bounds.sw().lng() < boundsWest){
467 					boundsWest=bounds.sw().lng();
468 				}
469 				if(bounds.ne().lat() > boundsNorth){
470 					boundsNorth=bounds.ne().lat();
471 				}
472 				if(bounds.ne().lng() > boundsEast){
473 					boundsEast=bounds.ne().lng();
474 				}
475 			}
476 		}
477                 //return line bounds
478                 var sw = new khtml.maplib.LatLng(boundsSouth, boundsWest);
479                 var ne = new khtml.maplib.LatLng(boundsNorth, boundsEast);
480                 var b = new khtml.maplib.geometry.Bounds(sw, ne);
481                 return (b);
482 
483 	}

492 		return (b);

511 	function getInternetExplorerVersion() {
512 		var rv = -1; // Return value assumes failure.
513 		if (navigator.appName == 'Microsoft Internet Explorer') {
514 			var ua = navigator.userAgent;
515 			var re = new RegExp("MSIE ([0-9]{1,}[\.0-9]{0,})");
516 			if (re.exec(ua) != null)
517 				rv = parseFloat(RegExp.$1);
518 		}
519 		return rv;
520 	}
