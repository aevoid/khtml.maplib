 23 /**
 24  * Event: Key is pressed.
 25  * Prefer 'keypress' over 'keydown' because 'keypress' allows to check which character was typed.
 26  * Because 'a' != 'A'.
 27  * We need this for the '-' and '+' signs (evt.which or evt.charCode).
 28  * For the arrow keys it does not matter - thus there is 'evt.keyCode' used.
 29  * 
 30  * @see http://www.brain4.de/programmierecke/js/tastatur.php
 31  * @see http://www.quirksmode.org/js/keys.html
 32  */
 33 khtml.maplib.base.Map.prototype.keypress = function(evt) {
 34 	var handled = true;
 35 	
 36 	switch (evt.keyCode) {
 37 	// arrow left
 38 	case 37:
 39 		this.moveit(30, 0);
 40 		break;
 41 	// arrow up
 42 	case 38:
 43 		this.moveit(0, 30);
 44 		break;
 45 	// arrow right
 46 	case 39:
 47 		this.moveit(-30, 0);
 48 		break;
 49 	// arrow down
 50 	case 40:
 51 		this.moveit(0, -30);
 52 		break;
 53 	default:
 54 		handled = false;
 55 		break;
 56 	}
 57 	if(handled) return true;
 58 	
 59 	var characterCode;	
 60 	if(typeof(evt.which)!='undefined') {
 61 		characterCode = evt.which;
 62 	} else {
 63 		characterCode = evt.charCode;
 64 	}
 65 	// Tested on IE8
 66 	if(this.internetExplorer) {
 67 		characterCode = evt.keyCode;
 68 	}
 69 	
 70 	switch (characterCode) {
 71 	case 43: // plus '+'
 72 		var dz = Math.ceil(this.getZoom() + 0.01) - this.getZoom();
 73 		this.autoZoomIn(this.mapsize.width / 2, this.mapsize.height / 2, dz);
 74 		break;
 75 	case 45: // minus '-'
 76 		var dz = Math.floor(this.getZoom() - 0.01) - this.getZoom();
 77 		this.autoZoomIn(this.mapsize.width / 2, this.mapsize.height / 2, dz);
 78 		break;
 79 	}
 80 	
 81 	return true;
 82 }

 89 /**
 90  * Call moveitexec 20 times to prepare the timeout's.
 91  */
 92 khtml.maplib.base.Map.prototype.moveit = function(x, y) {
 93 	var steps = 20;
 94 	var dx = x / steps;
 95 	var dy = y / steps;
 96 	for ( var i = 0; i < steps; i++) {
 97 		var f = Math.cos(3 * (-steps / 2 + i) / steps) * 5;
 98 		this.moveitexec(dx * f, dy * f, i);
 99 	}
100 }
101 /**
102  * Execute "moveXY" delayed by a timeout.
103  */
104 khtml.maplib.base.Map.prototype.moveitexec = function(dx, dy, i) {
105 	var that = this;
106 	var tempFunc = function() {
107 		that.moveXY(dx, dy);
108 	}
109 	setTimeout(tempFunc, 20 * i);
110 }
