1 khtml.maplib.parser.Osm=function(){
  2 	this.featureCollection=new khtml.maplib.overlay.FeatureCollection();
  3 	this.nodes=new Array();
  4 	this.ways=new Array();
  5 	this.data=function(){
  6 		return this.featureCollection;
  7 	}
  8 	this.add=function(dom){
  9 		var nodes= dom.getElementsByTagName("node")
 10 		for(var i=0;i<nodes.length;i++){
 11 			var node=nodes[i];
 12 			var id=node.getAttribute("id");
 13 			if(this.nodes[id]){
 14 				//node is already in the array
 15 			}else{
 16 				var tags=node.getElementsByTagName("tag");
 17 				var point=new khtml.maplib.LatLng(node.getAttribute("lat"),node.getAttribute("lon"));
 18 				this.nodes[id]=point;
 19 				var div=document.createElement("div");
 20 				if(tags.length >0){
 21 					for(var t=0;t<tags.length;t++){
 22 						var tag=tags[t];
 23 						var k=tag.getAttribute("k");
 24 						var v=tag.getAttribute("v");
 25 						div.setAttribute("mapcss_"+k,v);
 26 						//div.style.width="16px";
 27 						//div.style.height="16px";
 28 					}
 29 					
 30 					var marker=new khtml.maplib.overlay.Marker(point,div);
 31 					this.featureCollection.appendChild(marker);
 32 				}
 33 			}
 34 		
 35 		}
 36 
 37 		var ways= dom.getElementsByTagName("way")
 38 		for(var w=0;w<ways.length;w++){
 39 			var way=ways[w];
 40 			var id=way.getAttribute("id");
 41 			if(this.ways[id]){
 42 				//way already in array
 43 			}else{
 44 				var line=new khtml.maplib.geometry.Feature("Polyline");
 45 				line.close=true;
 46 				line.id=id;
 47 				var nds=way.getElementsByTagName("nd");
 48 				for(var d=0;d<nds.length;d++){
 49 					var nd=nds[d];
 50 					var ref=nd.getAttribute("ref");
 51 					line.geometry.coordinates.push(this.nodes[ref]);
 52 				}
 53 				var tags=way.getElementsByTagName("tag");
 54 				for(var t=0;t<tags.length;t++){
 55 					var tag=tags[t];
 56 					var k=tag.getAttribute("k");
 57 					var v=tag.getAttribute("v");
 58 					line.properties[k]=v;
 59 				}
 60 				this.featureCollection.appendChild(line);
 61 				this.ways[id]=line;
 62 			}
 63 		}
 64 
 65 	}
 66 }
