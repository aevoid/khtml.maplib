1 /**
  2 @namespace
  3 Feature
  4 
  5 Call like this:
  6 var geometry={
  7 	type:"LineString",
  8 	coordinates:[[10,01],[20,20]],
  9 	properties:["name":"Bahnhofstrasse"]
 10 }
 11 
 12 khtml.maplib.geometry.Feature(geometry);
 13 
 14 or
 15 
 16 khtml.maplib.geometry.Feature({type:"LineString",coordinates:[[10,01],[20,20]],properties:["name":"Bahnhofstrasse"]}); //UTF-8!!!!!
 17 
 18 or
 19 
 20 var feature=khtml.maplib.geometry.Feature("LineString");
 21 feature.geometry.coordinates=khtml.maplib.helpers.parseLine([[10,01],[20,20]]);
 22 feature.properties.name="blabla";
 23 
 24 */
 25 
 26 
 27 khtml.maplib.geometry.Feature = function(geometry) {
 28 	this.type="Feature";
 29 	this.geometry=new Object;
 30 	this.geometry.coordinates=new Array();
 31 	this.properties=new Object();
 32 	this.events = new Object;
 33 	this.style = new Object;
 34 	this.className = new Object;
 35 	this.className.baseVal = "khtml_vector_line";
 36 	if(typeof(geometry)=="object"){
 37 		if(geometry.type){
 38 			if(geometry.type=="Feature"){
 39 				this.geometry.type=geometry.geometry.type;
 40 				this.geometry.coordinates=khtml.maplib.base.helpers.parseLine(geometry.geometry.coordinates);
 41 				if(geometry.className){
 42 					this.className.baseVal=geometry.className.baseVal;
 43 				};
 44 				if(geometry.geometry.type=="FeatureCollection"){
 45 					var fc=new khtml.maplib.overlay.FeatureCollection();
 46 
 47 					for(var i=0;i<geometry.features.length;i++){
 48 						var feature=geometry.features[i];
 49 						fc.appendChild(new khtml.maplib.geometry.Feature(feature));
 50 					}
 51 					return fc;	
 52 				}
 53 			}else{
 54 				this.geometry.type=geometry.type;
 55 				if(geometry.coordinates){
 56 					this.geometry.coordinates=khtml.maplib.base.helpers.parseLine(geometry.coordinates);
 57 				}
 58 			}
 59 		}
 60 	}
 61 	if(typeof(geometry)=="string"){ //ok why not
 62 		this.geometry.type=geometry;
 63 	}
 64 
 65 	this.setPoints=function(points){
 66 		this.geometry.coordinates=khtml.maplib.base.helpers.parseLine(points);
 67 	}
 68 
 69 
 70 }
